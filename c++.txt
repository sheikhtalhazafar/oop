// ascending and descending order print
#include <iostream>
#include<stdio.h>
using namespace std;

int main() {
    int a[5];
    int i,j, temp;
    cout<< "enter array";
    
    for(i=0; i<5; i++){
         cin>> a[i]; // 5 4 2 1 3
    }
    for(i=0; i<5; i++){
        
        for(j=i+1; j<5; j++){  //4
            if(a[i]>a[j]){  // i=5 > j=3
                temp = a[i]; // temp = 5
                a[i] = a[j]; // i = 3
                a[j] = temp; // j = 5
            }
        }
    }
    
     for(i=0; i<5; i++){
         cout<< a[i]; // 5 4 2 1 3
    }
    
}
